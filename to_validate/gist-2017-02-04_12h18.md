--name "$CTN_traefik" \
--network "$NTW_FRONT" \
--network "$NTW_MONIT" \
--publish 80:80 --publish 443:443 --publish "$GUI_PORT_traefik":8080 \
--mount type=bind,source=/var/run/docker.sock,target=/var/run/docker.sock \
--mount type=bind,source=/somepath/traefik/traefik.toml,target=/etc/traefik/traefik.toml \

--label traefik.enable=false \
